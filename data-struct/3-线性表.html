    # 2.1 线性表的定义

    1）定义：线性表是具有相同特性的数据元素的一个有限序列

    • 数据元素：（a1, a2, ...ai-1, ai, ai+1, ... ,an）
    • 第一个元素a1称为线性起点，最后一个元素an称为线性终点
    • ai-1是ai的直接前驱，ai+1是ai的直接后继
    • 下标：1~n，是元素的序号，表示元素在表中的位置
    • n为元素总个数，即表长。n = 0时称为空表

    2）线性表的特征

    • 在非空的线性表，有且仅有一个开始结点a1,它没有直接前趋，而仅有一个直接后继a2;
    • 有且仅有一个终端结点an,它没有直接后继，而仅有一个直接前趋an-1;
    • 其余的内部结点ai (2<=i<=n-1) 都有且仅有一个直接前趋ai-1和一个直接后继ai+1。

    # 2.2 线性表的顺序存储表示

    1）定义：线性表的顺序表示又称为为顺序存储结构或顺序映像

    2）顺序存储定义：把逻辑上相邻的元素存储在物理上相邻的存储单元中的存储结构（简而言之，逻辑上相邻，物理上也相邻）

    3）基地址：线性表的第1个数据元素a1的存储位置，称作线性表的起始位置或基地址

    4）计算某个地址存储位置：LOC(ai) = LOC(a1)(基地址)+ (i - 1) x l

    线性表顺序存储结构占用一片连续的存储空间。只要知道某个元素的存储位置就可以计算其他元素的存储位置

    # 2.3.1 顺序表的代码实现

    // Created by Mr Cheng

#include <iostream>
#define MAXSIZE 100
using namespace std;

//定义存储结构
typedef int ElemType;
typedef struct {
    ElemType *elem;
    int length;
}SqList;

//1.初始化
bool InitList_S(SqList &L)
{
    L.elem = new ElemType [MAXSIZE];
    if (!L.elem) return false; //分配失败
    L.length = 0;
    return true;
}

//2.销毁
bool DestoryList(SqList &L)
{
    if (L.elem) //释放存储空间
    {
        delete L.elem;
        return true;
    }
    return false;
}

//3.清空
void ClearList(SqList &L)
{
    L.length = 0; //长度置0
}

//4.求表长
int GetLength(SqList &L)
{
    return L.length;
}

//5.判断表是否为空
bool IsEmpty(SqList &L)
{
    if (!L.length) return true;
    return false;
}

//6.顺序表的取值(根据位置i获取相应位置数据元素的内容)
bool GetElem(SqList L, int i, ElemType &e)
{
    if (i < 1 || i > L.length) return false; //i值不合法
    e = L.elem[i - 1];
    return true;
}

//7.按值查找
int LocateElem_S(SqList L, int e)
{
    int i = 0;
    while (i < L.length && L.elem[i] != e) i++;
    if (i < L.length) return i + 1; //查找成功返回序号
    return false; //查找失败返回0
}

//8.插入
bool ListInsert_S(SqList &L, int k, int e)
{
    if (L.length == MAXSIZE || k < 1) return false;
    for (int i = L.length - 1; i > k - 1; i--)
        L.elem[i + 1] = L.elem[i];
    L.elem[k - 1] = e;
    L.length++;
    return true;
}

//9.按序号删除
bool ListDelete(SqList &L, int k)
{
    if (k < 1 || k > L.length) return false;
    for (int i = k; i < L.length; i++)
        L.elem[i - 1] = L.elem[i];
    L.length--;
    return true;
}

//10.按值删除
bool ListDeleteByElem(SqList &L, int e)
{
    ListDelete(L, LocateElem_S(L, e));
}

//11.打印
void Print(SqList L)
{
    if (!L.length) {
        cout << "顺序表为空!" << endl;
        return;
    }
    cout << L.elem[0];
    for (int i = 1; i < L.length; i++)
        cout << "->" << L.elem[i];
    cout << endl;
}

int main()
{
    int i, s, e, choose;
    SqList L;
    cout << "1. 初始化\n";
    cout << "2. 插入\n";
    cout << "3. 查找\n";
    cout << "4. 取值\n";
    cout << "5. 按值删除\n";
    cout << "6. 按位置删除\n";
    cout << "7. 求表长\n";
    cout << "8. 清空表\n";
    cout << "9. 销毁表\n";
    cout << "10. 输出\n";
    cout << "0. 退出\n";
    choose = -1;
    while(choose != 0)
    {
        cout << "请输入数字选择：";
        cin >> choose;
        switch (choose)
        {
            case 1: //初始化
                if(InitList_S(L))
                    cout << "初始化成功！\n";
                break;
            case 2: //插入
            {
                cout << "请输入要插入的位置：";
                cin >> i;
                cout << "请输入要插入的值：";
                cin >> e;
                if(ListInsert_S(L, i, e))//插入
                    cout << "插入成功" << endl;
                else
                    cout << "插入失败" << endl;
                cout << "新的顺序表" << endl;
                Print(L);
                break;
            }
            case 3: // 查找
            {
                cout << "请输入要查找的值:";
                cin >> s;
                int ans = LocateElem_S(L, s);
                if (ans)//查找
                    cout << "所在序号为:" << ans << endl;
                break;
            }
            case 4: // 取值
            {
                cout << "请输入位置序号:";
                cin >> s;
                if (GetElem(L, s, e)) //取值
                    cout << e << endl;
                break;
            }
            case 5: //按值删除
            {
                cout << "请输入要删除的值：";
                cin >> e;
                if (ListDeleteByElem(L, e)) //按值删除
                    cout << "删除成功" << endl;
                else cout << "删除失败" << endl;
                cout << "新的顺序表" << endl;
                Print(L);
                break;
            }
            case 6://按位置删除
            {
                int i;
                cout << "请输入要删除的位置：";
                cin >> i;
                if(ListDelete(L, i)) //单链表的删除
                    cout << "删除成功" << endl;
                else cout << "删除失败" << endl;
                cout << "新的链表" << endl;
                Print(L);
                break;
            }
            case 7: //求表长
            {
                cout << "顺序表的长度:" << GetLength(L) << endl;
                break;
            }
            case 8: //清空表
            {
                ClearList(L);
                cout << "清空成功!" << endl;
            }
            case 9: //销毁
            {
                if (DestoryList(L))
                    cout << "销毁成功!" << endl;
                else
                    cout << "销毁失败!" << endl;
                break;
            }
            case 10: //输出
            {
                Print(L);
                break;
            }
            default: choose = 0;
        }
    }
    return 0;
}

    # 2.3.2 顺序表的操作算法分析

    • 时间复杂度
      • 查找、插入、删除算法的平均时间复杂度为O(n)
    • 空间复杂度
      • S(n) = O(1)

    # 2.3.3 顺序表的特优缺点

    1）特点

    • 以物理位置相邻表示逻辑关系

    2）优点

    • 存储密度大(结点本身所占存储量/结点结构所占存储量)
    • 可以随机存取表中的任一元素

    3）缺点

    • 在插入、删除某一元素时，需要移动大量元素
    • 浪费存储空间
    • 属于静态存储形式，数据元素的个数不能自由扩充

    # 2.3 链表

    # 2.3.1 线性表的链式表示和实现

    • 用一组物理位置任一的存储单元来存放线性表的数据元素
    • 这组存储单元既可以是连续的，也可以是不连续的，甚至是零散分布在内存中的任意位置上

    # 2.3.2 单链表的代码实现

    #include <iostream>
#include <string>
using namespace std;

typedef struct LNode{
    int data;	//结点数据域
    struct LNode *next;	//结点指针域
}LNode, *LinkList;	//LinkList为指向结构体LNode的指针类型

//1.初始化
bool InitList_L(LinkList &L)
{
    L = new LNode;	//生成新结点作为头结点， 用头指针L指定头结点
    if(!L) return false;	//生成结点失败
    L->next = NULL;	//头结点的指针域置空
    return true;
}

//2.前插法创建单链表
void CreateList_H(LinkList &L)
{
    int n;
    LinkList s;	//定义一个指针变量
    L = new LNode;
    L->next = NULL; //建立一个带头结点的空链表
    cout << "请输入元素个数n：" << endl;
    cin >> n;
    cout << "请依次输入n个元素：" << endl;
    cout << "前插法创建单链表..." << endl;
    while(n--)
    {
        s = new LNode; //生成新结点s
        cin >> s->data;	//输入元素值赋给新结点的数据域
        s->next = L->next;
        L->next = s; //将新结点s插入到头结点之后
    }

}

//3.尾插法创建单链表
void CreateList_R(LinkList &L)
{
    int n;
    LinkList s, r;	//定义一个指针变量
    L = new LNode;
    L->next = NULL; //建立一个带头结点的空链表
    r = L; //尾指针r指向头结点
    cout << "请输入元素个数n：" << endl;
    cin >> n;
    cout << "请依次输入n个元素：" << endl;
    cout << "尾插法创建单链表..." << endl;
    while(n--)
    {
        s = new LNode; //生成新结点s
        cin >> s->data;	//输入元素值赋给新结点的数据域
        s->next = NULL;
        r->next = s; //将新结点s插入尾结点*r之后
        r = s; //r指向新的尾结点s
    }

}

//4.单链表的取值
bool GetElem_L(LinkList L, int i, int &e)
{
    //在头结点的单链表L中查找第i个元素
    //用e记录L中第i个数据元素的值
    int j;
    LinkList p;
    p = L->next;//p指向第一个结点
    j = 1; //j为计数器
    while(j < i && p) //沿链域向后扫描， 直到p指向第i个元素或p为空
    {
        j++;
        p = p->next;
    }
    if(!p || j > i) return false; //i值不合法i>j或者i<=0
    e = p->data;	//取第i个结点的数据域
    return true;
}

//5.按值查找
bool LocateElem_L(LinkList L, int e)
{
    //在带头结点的单链表L中查找值为e的元素
    LinkList p;
    p = L->next;
    while(p && p->data != e) //沿链域向后扫描，直到p为空或p所指结点的数据域等
        p = p->next; //p指向下一个结点
    if(!p) return false; //查找失败p为null
    return true;
}

//6.单链表的插入
bool ListInsert_L(LinkList &L, int i, int e)
{
    //在带头结点的单链表L中第i个位置插入为e的新结点
    int j;
    LinkList p, s;
    p = L;
    j = 0;
    while(p && j < i - 1) //查找第i-1个结点，p指向该结点
    {
        p = p->next;
        j++;
    }
    if(!p || j > i + 1) return false; //i > n+1或者i<1
    s = new LNode;  //生成新结点
    s->data = e;	//将新结点的数据域置为e
    s->next = p->next;	//将新结点的指针域指向结点ai
    p->next = s;	//将结点p的指针域指向结点s
    return true;
}

//7.单链表的删除
bool ListDelete_L(LinkList &L, int i)
{
    //在带头结点的单链表L中，删除第i个位置
    LinkList p, q;
    int j;
    p = L;
    j = 0;
    while((p->next) && (j < i - 1)) //查找第i-1个结点，p指向该结点
    {
        p = p->next;
        j++;
    }
    if(!(p->next) || (j > i + 1)) //当i>n或i<1时，删除位置不合理
        return false;
    q = p->next;	//临时保存被删结点的地址以备释放空间
    p->next = q->next;	//改变删除结点前驱结点的指针域
    delete q; //释放被删除节点的空间
    return true;
}

//8.单链表的输出
void LinkPrint_L(LinkList L)
{
    LinkList p;
    p = L->next;
    while(p)
    {
        cout << p->data << "\t";
        p = p->next;
    }
    cout << endl;
}
int main()
{
    int i, x, e, choose;
    LinkList L;
    cout << "1. 初始化\n";
    cout << "2. 创建单链表 (前插法)\n";
    cout << "3. 创建单链表 (尾插法)\n";
    cout << "4. 取值\n";
    cout << "5. 查找\n";
    cout << "6. 插入\n";
    cout << "7. 删除\n";
    cout << "8. 输出\n";
    cout << "0. 退出\n";
    choose = -1;
    while(choose != 0)
    {
        cout << "请输入数字选择：";
        cin >> choose;
        switch (choose)
        {
            case 1: //初始化一个空的单链表
                if(InitList_L(L))
                    cout << "初始化一个空的单链表！\n";
                break;
            case 2: //创建单链表(前插法)
                CreateList_H(L);
                cout << "前插法创建单链表输出结果:\n";
                LinkPrint_L(L);
                break;
            case 3: //创建单链表(尾插法)
                CreateList_R(L);
                cout << "前插法创建单链表输出结果:\n";
                LinkPrint_L(L);
                break;
            case 4: // 取值
            {
                int s, e;
                cout << "请输入:";
                cin >> s;
                GetElem_L(L, s, e); //单链表的取值
                cout << e << endl;
                break;
            }
            case 5: //查找
            {
                int e;
                cout << "请输入要查找的值：";
                cin >> e;
                if (LocateElem_L(L, e)) //按值查找
                    cout << "找到了" << endl;
                else cout << "未找到" << endl;
                break;
            }
            case 6: //插入
            {
                int i, e;
                cout << "请输入要插入的位置：";
                cin >> i;
                cout << "请输入要插入的值：";
                cin >> e;
                if(ListInsert_L(L, i, e))//单链表的插入
                    cout << "插入成功" << endl;
                else cout << "插入失败" << endl;
                cout << "新链表" << endl;
                LinkPrint_L(L);
                break;
            }
            case 7://删除
            {
                int i;
                cout << "请输入要删除的位置：";
                cin >> i;
                if(ListDelete_L(L, i)) //单链表的删除
                    cout << "删除成功" << endl;
                else cout << "删除失败" << endl;
                cout << "新的链表" << endl;
                LinkPrint_L(L);
                break;
            }

            case 8: //输出
                LinkPrint_L(L);
                break;
            default: choose = 0;
        }
    }
    return 0;
}

    # 2.10 顺序表和链表的比较

    1）存取方式

    • 顺序表可以顺序存取，也可以随机存取
    • 链表只能从表头顺序存取元素

    2）逻辑结构与物理结构

    • 顺序存储时，逻辑上相邻的元素，对应的物理存储位置也相邻
    • 链式存储时，逻辑上相邻的元素，物理存储位置不一定相邻，其逻辑关系是通过指针链接来表示的

    3）查找、插入和删除操作

    • 对于按值查找，顺序表无序时，两者的时间复杂度均为 O(n)；顺序表有序时，可采用折半查找，此时的时间复杂度为 O(log2n)

      对于按序号查找，顺序表支持随机访问，时间复杂度仅为 O(1)，而链表的平均时间复杂度为 O(n)。

    • 顺序表的插入、删除操作，平均需要移动半个表长的元素

      链表的插入、删除操作，只需要修改相关结点的指针域既可。由于链表的每个结点都有指针域，所以在存储空间上要比顺序表付出的代价大，存储密度不够大。

    4）空间分配

    • 顺序存储在静态存储分配情况下，一旦存储空间装满就不能扩充，若再加入新元素，则会出现溢出，所以需要预先分配足够大的存储空间。预先分配过大，可能会导致顺序表候补大量闲置；预先分配过小，又会造成溢出。

      动态存储分配虽然存储空间可以扩充，但需要移动大量元素，导致操作效率降低，而且若内存中没有更大的连续存储空间，则会导致分配失败

    • 链表只在需要时申请分配，高效灵活
